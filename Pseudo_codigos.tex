\begin{algorithm}
\begin{algorithmic}[1]
  \REQUIRE $n$
  \ENSURE  $suma$
  \STATE $suma \leftarrow 0$
  \FOR{$i\gets 1,n$}
  \STATE $suma\leftarrow suma+i^2$
  \ENDFOR
  \RETURN $suma$
\end{algorithmic} 
  
  \caption{BISECTION}
  \label{a1}
\end{algorithm}

\begin{algorithm}
  \begin{algorithmic}[0]
    \REQUIRE $n$
    \ENSURE  $suma$
    \STATE $suma \leftarrow 0$
    \STATE $i \leftarrow 1$
    \WHILE{$i\leq n$}
    \STATE $suma\leftarrow suma+i^2$
    \STATE $i \leftarrow i+1$
    \ENDWHILE
    \RETURN $suma$
  \end{algorithmic}
  \caption{Uso del entorno While}
  \label{a2}
\end{algorithm}
